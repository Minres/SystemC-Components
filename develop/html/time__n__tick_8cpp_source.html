66 inline void time2tick::clocker() {
67  while(true) {
68  auto t = clk_i.read();
69  if(t == sc_core::SC_ZERO_TIME) {
70  wait(clk_i.value_changed_event());
71  t = clk_i.read();
72  }
73  clk_o = true;
74  wait(t / 2);
75  clk_o = false;
76  wait(t - t / 2);
77  }
