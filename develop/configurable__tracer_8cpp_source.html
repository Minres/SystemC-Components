52 void configurable_tracer::descend(const sc_core::sc_object* obj, bool trace) {
53  if(obj == this)
54  return;
55  const std::string kind = obj->kind();
56  if((types_to_trace & trace_types::SIGNALS) == trace_types::SIGNALS && kind == "tlm_signal") {
57  if(trace)
58  obj->trace(trf);
59  return;
60  } else if(kind == "sc_vector") {
61  if(trace)
62  for(auto o : obj->get_child_objects())
63  descend(o, trace);
64  return;
65  } else if(kind == "sc_module") {
66  auto trace_enable = get_trace_enabled(obj, default_trace_enable);
67  if(trace_enable)
68  obj->trace(trf);
69  for(auto o : obj->get_child_objects())
70  descend(o, trace_enable);
71  } else if(kind == "sc_variable") {
72  if(trace && (types_to_trace & trace_types::VARIABLES) == trace_types::VARIABLES)
73  obj->trace(trf);
74  } else if(kind == "sc_signal" || kind == "sc_clock" || kind == "sc_buffer" || kind == "sc_signal_rv") {
75  if(trace && (types_to_trace & trace_types::SIGNALS) == trace_types::SIGNALS)
76  try_trace(trf, obj, types_to_trace);
77  } else if(kind == "sc_in" || kind == "sc_out" || kind == "sc_inout") {
78  if(trace && (types_to_trace & trace_types::PORTS) == trace_types::PORTS)
79  try_trace(trf, obj, types_to_trace);
80  } else if(const auto* tr = dynamic_cast<const scc::traceable*>(obj)) {
81  if(tr->is_trace_enabled())
82  obj->trace(trf);
83  for(auto o : obj->get_child_objects())
84  descend(o, tr->is_trace_enabled());
85 // } else if(trace && traceable_kinds.find(kind)!=traceable_kinds.end()) {
86 // try_trace(trf, obj, types_to_trace);
87  }
88 }

90 auto scc::configurable_tracer::get_trace_enabled(const sc_core::sc_object* obj, bool fall_back) -> bool {
91  auto* attr = obj->get_attribute(EN_TRACING_STR);
92  if(attr != nullptr && dynamic_cast<const sc_core::sc_attribute<bool>*>(attr) != nullptr) {
93  const auto* a = dynamic_cast<const sc_core::sc_attribute<bool>*>(attr);
94  return a->value;
95  } else {
96  std::string hier_name{obj->name()};
97  auto h = cci_broker.get_param_handle(hier_name.append("." EN_TRACING_STR));
98  if(h.is_valid())
99  return h.get_cci_value().get_bool();
100  }
101  return fall_back;
102 }

104 void configurable_tracer::augment_object_hierarchical(sc_core::sc_object* obj) {
105  if(dynamic_cast<sc_core::sc_module*>(obj) != nullptr || dynamic_cast<scc::traceable*>(obj) != nullptr) {
106  auto* attr = obj->get_attribute(EN_TRACING_STR);
107  if(attr == nullptr ||
108  dynamic_cast<const sc_core::sc_attribute<bool>*>(attr) == nullptr) { // check if we have no sc_attribute
109  std::string hier_name{obj->name()};
110  if(hier_name.substr(0, 11) != "scc_tracer") {
111  hier_name += "." EN_TRACING_STR;
112  auto h = cci_broker.get_param_handle(hier_name);
113  if(!h.is_valid()) // we have no cci_param so create one
114  params.push_back(new cci::cci_param<bool>(hier_name, default_trace_enable, cci_broker, "", cci::CCI_ABSOLUTE_NAME,
115  cci_broker.get_originator()));
116  else
117  h.set_cci_value(cci::cci_value{default_trace_enable});
118  }
119  } else if(auto battr = dynamic_cast<sc_core::sc_attribute<bool>*>(attr)) {
120  battr->value = default_trace_enable;
121  }
122  for(auto* o : obj->get_child_objects())
123  augment_object_hierarchical(o);
124  }
