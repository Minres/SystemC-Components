351  assert(creator);
352  std::stringstream ss; ss << name << "(" << idx << ")";
353  ret=values.at(idx)=creator(ss.str().c_str(), idx);
354  }
355  return *ret;
356  }

359  assert(values.at(idx) && "No initialized value in sc_variable_vector position");
