364  void resize(size_t sz, T def_val) {
365  assert(!sc_core::sc_get_curr_simcontext()->elaboration_done());
366  values.resize(sz);
367  auto idx = 0U;
368  for(auto& e : values) {
369  std::stringstream ss;
370  ss << name << "(" << idx++ << ")";
371  e = new sc_variable<T>(ss.str().c_str(), def_val);
372  }
373  }
374 
375  bool is_valid(size_t idx) const { return values.size()> idx && values.at(idx) != nullptr; }
376 
377  sc_variable<T>& operator[](size_t idx) {
378  auto ret = values.at(idx);
379  if(!ret) {
380  assert(!sc_core::sc_get_curr_simcontext()->elaboration_done());
381  assert(creator);
382  std::stringstream ss;
383  ss << name << "(" << idx << ")";
384  ret = values.at(idx) = creator(ss.str().c_str(), idx);
385  }
386  return *ret;
387  }
388 
389  sc_variable<T> const& operator[](size_t idx) const {
390  assert(values.at(idx) && "No initialized value in sc_variable_vector position");
391  return *values.at(idx);
392  }
393  ~sc_variable_vector() {
394  for(auto p : values)
395  delete p;
396  }
