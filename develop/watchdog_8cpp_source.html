34 void watchdog::guard() {
35  while(live.load()) {
36  if(idle.load()) {
37  // Sleep indefinitely until either told to become active or destruct
38  std::unique_lock<std::mutex> live_lock(guard_mutex);
39  wakeup.wait(live_lock, [this]() { return !this->idle.load() || !this->live.load(); });
40  };
41  if(!live.load())
42  break;
43  // the actual timeout checking
44  auto now = system_clock::now();
45  if((now - touched.load()) > timeout) {
46  idle.store(true);
47  alarm_cb();
48  continue; // skip waiting for next timeout
49  }
50  {
51  // sleep until next timeout check or destruction
52  std::unique_lock<std::mutex> live_lock(guard_mutex);
53  wakeup.wait_for(live_lock, sleep_duration, [this]() { return !this->live.load(); });
54  }
55  };
56 }
