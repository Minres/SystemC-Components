37 template <typename T>
38 inline auto try_trace_obj(sc_trace_file* trace_file, const sc_object* object, trace_types types_to_trace) -> bool {
39  if((types_to_trace & trace_types::PORTS) == trace_types::PORTS) {
40  if(auto const* ptr = dynamic_cast<sc_core::sc_in<T> const*>(object)) {
41  SC_TRACE_NS sc_trace(trace_file, *ptr->get_interface(0), object->name());
42  return true;
43  }
44  if(auto const* ptr = dynamic_cast<sc_core::sc_inout<T> const*>(object)) {
45  SC_TRACE_NS sc_trace(trace_file, *ptr->get_interface(0), object->name());
46  return true;
47  }
48  }
49  if((types_to_trace & trace_types::SIGNALS) == trace_types::SIGNALS) {
50  if(auto const* ptr = dynamic_cast<sc_core::sc_signal_inout_if<T> const*>(object)) {
51  SC_TRACE_NS sc_trace(trace_file, *ptr, object->name());
52  return true;
53  }
54  }
55  return false;
56 }

58 template <size_t SIZE> struct ForLoop {
59  template <template <size_t> class Func>
60  static bool iterate(sc_trace_file* trace_file, const sc_object* object, trace_types types_to_trace) {
61  if(ForLoop<SIZE - (SIZE > 128 ? 8 : 1)>::template iterate<Func>(trace_file, object, types_to_trace))
62  return true;
63  else
64  return Func<SIZE>()(trace_file, object, types_to_trace);
65  }
66 };

254 void tracer_base::descend(const sc_object* obj, bool trace_all) {
255  if(obj == this)
256  return;
257  const char* kind = obj->kind();
258  if(strcmp(kind, "tlm_signal") == 0) {
259  obj->trace(trf);
260  } else if(strcmp(kind, "sc_vector") == 0) {
261  for(auto o : obj->get_child_objects())
262  descend(o, trace_all);
263  } else if((strcmp(kind, "sc_module") == 0 && trace_all) || dynamic_cast<const traceable*>(obj)) {
264  obj->trace(trf);
265  for(auto o : obj->get_child_objects())
266  descend(o, trace_all);
267  } else if(strcmp(kind, "sc_variable") == 0) {
268  if((types_to_trace & trace_types::VARIABLES) == trace_types::VARIABLES)
269  obj->trace(trf);
270  } else if(const auto* tr = dynamic_cast<const scc::traceable*>(obj)) {
271  if(tr->is_trace_enabled())
272  obj->trace(trf);
273  for(auto o : obj->get_child_objects())
274  descend(o, tr->is_trace_enabled());
275  } else {
276  try_trace(trf, obj, types_to_trace);
277  }
278 }
