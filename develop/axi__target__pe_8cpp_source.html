110 void axi_target_pe::setup_callbacks(fsm_handle* fsm_hndl) {
111  fsm_hndl->fsm->cb[RequestPhaseBeg] = [this, fsm_hndl]() -> void {
112  fsm_hndl->beat_count = 0;
113  outstanding_cnt[fsm_hndl->trans->get_command()]++;
114  };
115  fsm_hndl->fsm->cb[BegPartReqE] = [this, fsm_hndl]() -> void {
116  if(!fsm_hndl->beat_count && max_outstanding_tx.value &&
117  outstanding_cnt[fsm_hndl->trans->get_command()] > max_outstanding_tx.value) {
118  stalled_tx[fsm_hndl->trans->get_command()] = fsm_hndl->trans.get();
119  stalled_tp[fsm_hndl->trans->get_command()] = EndPartReqE;
120  } else { // accepted, schedule response
121  getOutStandingTx(fsm_hndl->trans->get_command())++;
122  if(wr_data_accept_delay.get_value())
123  schedule(EndPartReqE, fsm_hndl->trans, wr_data_accept_delay.get_value() - 1);
124  else
125  schedule(EndPartReqE, fsm_hndl->trans, sc_core::SC_ZERO_TIME);
126  }
127  };
128  fsm_hndl->fsm->cb[EndPartReqE] = [this, fsm_hndl]() -> void {
129  tlm::tlm_phase phase = axi::END_PARTIAL_REQ;
130  sc_time t(clk_if ? ::scc::time_to_next_posedge(clk_if) - 1_ps : SC_ZERO_TIME);
131  auto ret = socket_bw->nb_transport_bw(*fsm_hndl->trans, phase, t);
132  fsm_hndl->beat_count++;
133  };
134  fsm_hndl->fsm->cb[BegReqE] = [this, fsm_hndl]() -> void {
135  if(!fsm_hndl->beat_count && max_outstanding_tx.value &&
136  outstanding_cnt[fsm_hndl->trans->get_command()] > max_outstanding_tx.value) {
137  stalled_tx[fsm_hndl->trans->get_command()] = fsm_hndl->trans.get();
138  stalled_tp[fsm_hndl->trans->get_command()] = EndReqE;
139  } else { // accepted, schedule response
140  if(!fsm_hndl->beat_count)
141  getOutStandingTx(fsm_hndl->trans->get_command())++;
142  auto latency = fsm_hndl->trans->is_read() ? rd_addr_accept_delay.get_value() : wr_data_accept_delay.get_value();
143  if(latency)
144  schedule(EndReqE, fsm_hndl->trans, latency - 1);
145  else
146  schedule(EndReqE, fsm_hndl->trans, sc_core::SC_ZERO_TIME);
147  }
148  };
149  fsm_hndl->fsm->cb[EndReqE] = [this, fsm_hndl]() -> void {
150  tlm::tlm_phase phase = tlm::END_REQ;
151  sc_time t(clk_if ? ::scc::time_to_next_posedge(clk_if) - 1_ps : SC_ZERO_TIME);
152  auto ret = socket_bw->nb_transport_bw(*fsm_hndl->trans, phase, t);
153  fsm_hndl->trans->set_response_status(tlm::TLM_OK_RESPONSE);
154  if(auto ext3 = fsm_hndl->trans->get_extension<axi3_extension>()) {
155  ext3->set_resp(resp_e::OKAY);
156  } else if(auto ext4 = fsm_hndl->trans->get_extension<axi4_extension>()) {
157  ext4->set_resp(resp_e::OKAY);
158  } else if(auto exta = fsm_hndl->trans->get_extension<ace_extension>()) {
159  exta->set_resp(resp_e::OKAY);
160  } else
161  sc_assert(false && "No valid AXITLM extension found!");
162  if(fw_o.get_interface())
163  fw_o->transport(*(fsm_hndl->trans));
164  else {
165  auto latency = operation_cb ? operation_cb(*fsm_hndl->trans)
166  : fsm_hndl->trans->is_read() ? rd_resp_delay.get_value() : wr_resp_delay.get_value();
167  if(latency < std::numeric_limits<unsigned>::max()) {
168  if(fsm_hndl->trans->is_write())
169  wr_req2resp_fifo.push_back(std::make_tuple(fsm_hndl->trans.get(), latency));
170  else if(fsm_hndl->trans->is_read())
171  rd_req2resp_fifo.push_back(std::make_tuple(fsm_hndl->trans.get(), latency));
172  }
173  }
174  };
175  fsm_hndl->fsm->cb[BegPartRespE] = [this, fsm_hndl]() -> void {
176  // scheduling the response
177  if(fsm_hndl->trans->is_read()) {
178  if(!rd_resp_beat_fifo.nb_write(std::make_tuple(fsm_hndl, BegPartRespE)))
179  SCCERR(SCMOD) << "too many outstanding transactions";
180  } else if(fsm_hndl->trans->is_write()) {
181  if(!wr_resp_beat_fifo.nb_write(std::make_tuple(fsm_hndl, BegPartRespE)))
182  SCCERR(SCMOD) << "too many outstanding transactions";
183  }
184  };
185  fsm_hndl->fsm->cb[EndPartRespE] = [this, fsm_hndl]() -> void {
186  fsm_hndl->trans->is_read() ? rd_resp_ch.post() : wr_resp_ch.post();
187  auto size = get_burst_lenght(*fsm_hndl->trans) - 1;
188  fsm_hndl->beat_count++;
189  if(rd_data_beat_delay.get_value())
190  schedule(fsm_hndl->beat_count < size ? BegPartRespE : BegRespE, fsm_hndl->trans, rd_data_beat_delay.get_value());
191  else
192  schedule(fsm_hndl->beat_count < size ? BegPartRespE : BegRespE, fsm_hndl->trans, SC_ZERO_TIME, true);
193  };
194  fsm_hndl->fsm->cb[BegRespE] = [this, fsm_hndl]() -> void {
195  // scheduling the response
196  if(fsm_hndl->trans->is_read()) {
197  if(!rd_resp_beat_fifo.nb_write(std::make_tuple(fsm_hndl, BegRespE)))
198  SCCERR(SCMOD) << "too many outstanding transactions";
199  } else if(fsm_hndl->trans->is_write()) {
200  if(!wr_resp_beat_fifo.nb_write(std::make_tuple(fsm_hndl, BegRespE)))
201  SCCERR(SCMOD) << "too many outstanding transactions";
202  }
203  };
204  fsm_hndl->fsm->cb[EndRespE] = [this, fsm_hndl]() -> void {
205  fsm_hndl->trans->is_read() ? rd_resp_ch.post() : wr_resp_ch.post();
206  if(rd_resp.get_value() < rd_resp.get_capacity()) {
207  SCCTRACE(SCMOD) << "finishing exclusive read response for trans " << *fsm_hndl->trans;
208  rd_resp.post();
209  }
210  auto cmd = fsm_hndl->trans->get_command();
211  outstanding_cnt[cmd]--;
212  getOutStandingTx(cmd)--;
213  if(cmd == tlm::TLM_READ_COMMAND)
214  active_rdresp_id.erase(axi::get_axi_id(fsm_hndl->trans.get()));
215  if(stalled_tx[cmd]) {
216  auto* trans = stalled_tx[cmd];
217  auto latency = trans->is_read() ? rd_addr_accept_delay.get_value() : wr_data_accept_delay.get_value();
218  if(latency)
219  schedule(stalled_tp[cmd], trans, latency - 1);
220  else
221  schedule(stalled_tp[cmd], trans, sc_core::SC_ZERO_TIME);
222  getOutStandingTx(cmd)++;
223  stalled_tx[cmd] = nullptr;
224  stalled_tp[cmd] = CB_CNT;
225  }
226  };
227 }

236 void axi::pe::axi_target_pe::process_req2resp_fifos() {
237  if(!rd_req2resp_fifo.empty()){
238  auto& entry = rd_req2resp_fifo.front();
239  if(std::get<1>(entry) == 0) {
240  if(rd_resp_fifo.num_free()){
241  rd_resp_fifo.write(std::get<0>(entry));
242  rd_req2resp_fifo.pop_front();
243  }
244  } else {
245  std::get<1>(entry) -= 1;
246  rd_req2resp_fifo.push_back(entry);
247  rd_req2resp_fifo.pop_front();
248  }
249  }
250  if(!wr_req2resp_fifo.empty()) {
251  auto& entry = wr_req2resp_fifo.front();
252  if(std::get<1>(entry) == 0) {
253  if(rd_resp_fifo.num_free()){
254  wr_resp_fifo.write(std::get<0>(entry));
255  wr_req2resp_fifo.pop_front();
256  }
257  } else {
258  std::get<1>(entry) -= 1;
259  wr_req2resp_fifo.push_back(entry);
260  wr_req2resp_fifo.pop_front();
261  }
262  }

265 void axi::pe::axi_target_pe::start_rd_resp_thread() {
266  auto residual_clocks = 0.0;
267  while(true) {
268  auto* trans = rd_resp_fifo.read();
269  if(!rd_data_interleaving.value || rd_data_beat_delay.get_value() == 0) {
270  while(!rd_resp.get_value())
271  wait(clk_i.posedge_event());
272  rd_resp.wait();
273  }
274  SCCTRACE(SCMOD) << __FUNCTION__ << " starting exclusive read response for trans " << *trans;
275  auto e = axi::get_burst_lenght(trans) == 1 || trans->is_write() ? axi::fsm::BegRespE : BegPartRespE;
276  auto id = axi::get_axi_id(trans);
277  while(active_rdresp_id.size() && active_rdresp_id.find(id) != active_rdresp_id.end()) {
278  wait(clk_i.posedge_event());
279  }
280  active_rdresp_id.insert(id);
281  if(rd_data_beat_delay.get_value())
282  schedule(e, trans, rd_data_beat_delay.get_value() - 1);
283  else
284  schedule(e, trans, SC_ZERO_TIME);
285  }
286 }

296 void axi::pe::axi_target_pe::send_rd_resp_beat_thread() {
297  std::tuple<fsm::fsm_handle*, axi::fsm::protocol_time_point_e> entry;
298  while(true) {
299  // waiting for responses to send
300  wait(rd_resp_beat_fifo.data_written_event());
301  while(rd_resp_beat_fifo.nb_read(entry)) {
302  // there is something to send
303  auto fsm_hndl = std::get<0>(entry);
304  auto tp = std::get<1>(entry);
305  sc_time t;
306  tlm::tlm_phase phase{tp == BegPartRespE ? axi::BEGIN_PARTIAL_RESP : tlm::tlm_phase(tlm::BEGIN_RESP)};
307  // wait to get ownership of the response channel
308  while(!rd_resp_ch.get_value())
309  wait(clk_i.posedge_event());
310  rd_resp_ch.wait();
311  SCCTRACE(SCMOD) << __FUNCTION__ << " starting exclusive read response for trans " << *fsm_hndl->trans;
312  if(socket_bw->nb_transport_bw(*fsm_hndl->trans, phase, t) == tlm::TLM_UPDATED) {
313  schedule(phase == tlm::END_RESP ? EndRespE : EndPartRespE, fsm_hndl->trans, 0);
314  }
315  }
316  }
317 }
