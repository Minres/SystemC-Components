126 template <typename T> inline void range_lut<T>::addEntry(T i, uint64_t base_addr, uint64_t size) {
127  auto iter = m_lut.find(base_addr);
128  if(iter != m_lut.end() && iter->second.index != null_entry)
129  throw std::runtime_error("range already mapped");
130 
131  auto eaddr = base_addr + size - 1;
132  if(eaddr < base_addr)
133  throw std::runtime_error("address wrap-around occurred");
134 
135  m_lut[base_addr] = lut_entry{i, size > 1 ? BEGIN_RANGE : SINGLE_BYTE_RANGE};
136  if(size > 1)
137  m_lut[eaddr] = lut_entry{i, END_RANGE};
138  ++m_size;
139 }

141 template <typename T> inline bool range_lut<T>::removeEntry(T i) {
142  auto start = m_lut.begin();
143  while(start->second.index != i && start != m_lut.end())
144  start++;
145  if(start != m_lut.end()) {
146  if(start->second.type == SINGLE_BYTE_RANGE) {
147  m_lut.erase(start);
148  } else {
149  auto end = start;
150  end++;
151  end++;
152  m_lut.erase(start, end);
153  }
154  --m_size;
155  return true;
156  }
157  return false;
158 }

160 template <typename T> inline void range_lut<T>::validate() const {
161  auto mapped = false;
162  for(auto iter = m_lut.begin(); iter != m_lut.end(); iter++) {
163  switch(iter->second.type) {
164  case SINGLE_BYTE_RANGE:
165  if(iter->second.index != null_entry && mapped)
166  throw std::runtime_error("range overlap: begin range while in mapped range");
167 
168  break;
169  case BEGIN_RANGE:
170  if(iter->second.index != null_entry) {
171  if(mapped) {
172  throw std::runtime_error("range overlap: begin range while in mapped range");
173  }
174  mapped = true;
175  }
176  break;
177  case END_RANGE:
178  if(!mapped) {
179  throw std::runtime_error("range overlap: end range while in unmapped region");
180  }
181  mapped = false;
182  break;
183  }
184  }
185 }

199  << iter->first << std::dec << " as " << iter->second->index << std::endl;
200  }
201  }
202  return buf.str();
