120 template <unsigned int DATA_WIDTH, unsigned int ADDR_WIDTH, unsigned int ID_WIDTH, unsigned int USER_WIDTH>
121 inline void target<DATA_WIDTH, ADDR_WIDTH, ID_WIDTH, USER_WIDTH>::target::clk_cb() {
122  sc_core::sc_time delay = sc_core::SC_ZERO_TIME;
123  if(clk_i.event()) {
124  if(rchannel_pending_rsp.size()) {
125  auto& head = rchannel_pending_rsp.front();
126  if(std::get<1>(head) == 0) {
127  rchannel_rsp.notify(std::get<0>(head));
128  rchannel_pending_rsp.pop_front();
129  }
130  for(auto& e : rchannel_pending_rsp) {
131  if(std::get<1>(e))
132  --std::get<1>(e);
133  }
134  }
135  }

138 template <unsigned int DATA_WIDTH, unsigned int ADDR_WIDTH, unsigned int ID_WIDTH, unsigned int USER_WIDTH>
139 inline tlm::tlm_sync_enum
140 target<DATA_WIDTH, ADDR_WIDTH, ID_WIDTH, USER_WIDTH>::target::nb_transport_bw(payload_type& trans, phase_type& phase,
141  sc_core::sc_time& t) {
142  auto id = obi::get_obi_id(trans);
143  auto* ext = trans.get_extension<obi::obi_extension>();
144  sc_assert(ext && "obi_extension missing");
145  switch(phase) {
146  case tlm::END_REQ: {
147  auto it = states.find(&trans);
148  sc_assert(it != states.end());
149  it->second.last_phase = tlm::END_REQ;
150  achannel_rsp.notify(&trans, t);
151  return tlm::TLM_ACCEPTED;
152  }
153  case tlm::BEGIN_RESP: {
154  auto it = states.find(&trans);
155  sc_assert(it != states.end());
156  auto& state = it->second;
157  if(state.pending_tx) {
158  unsigned resp_delay = addr2data_delay < 0 ? scc::MT19937::uniform(0, -addr2data_delay) : addr2data_delay;
159  if(resp_delay) {
160  rchannel_pending_rsp.push_back({state.pending_tx, resp_delay - 1});
161  } else
162  rchannel_pending_rsp.push_back({state.pending_tx, 0});
163  }
164  state.last_phase = tlm::BEGIN_RESP;
165  return tlm::TLM_ACCEPTED;
166  }
167  default:
168  SCCWARN(SCMOD) << phase << " is unsupported phase transaction combination";
169  return tlm::TLM_ACCEPTED;
170  }

173 template <unsigned int DATA_WIDTH, unsigned int ADDR_WIDTH, unsigned int ID_WIDTH, unsigned int USER_WIDTH>
174 inline void target<DATA_WIDTH, ADDR_WIDTH, ID_WIDTH, USER_WIDTH>::achannel_req_t() {
175  wait(SC_ZERO_TIME);
176  wait(clk_i.posedge_event());
177  while(true) {
178  while(resetn_i.read() == false)
179  wait(clk_i.posedge_event());
180  while(resetn_i.read() == true) {
181  gnt_o.write(req2gnt_delay == 0);
182  do {
183  wait(this->req_i.posedge_event() | clk_i.negedge_event());
184  } while(this->req_i.read() == false);
185  auto data_len = DATA_WIDTH / 8;
186  tlm::scc::tlm_gp_shared_ptr gp = tlm::scc::tlm_mm<>::get().allocate<obi::obi_extension>(data_len);
187  gp->set_streaming_width(data_len);
188  gp->set_address(addr_i.read());
189  gp->set_command(we_i.read() ? tlm::TLM_WRITE_COMMAND : tlm::TLM_READ_COMMAND);
190  auto be = static_cast<unsigned>(be_i.read());
191  auto cnt = util::bit_count(be);
192  gp->set_streaming_width(cnt);
193  gp->set_data_length(cnt);
194  if(we_i.read()) {
195  auto bus_data = wdata_i.read();
196  auto tx_byte_idx = 0;
197  for(auto i = 0U, be_idx = 0U; i < DATA_WIDTH; i += 8, ++be_idx) {
198  if(be & (1U << be_idx)) {
199  *(gp->get_data_ptr() + tx_byte_idx) = bus_data.range(i + 7, i).to_uint();
200  tx_byte_idx++;
201  }
202  }
203  SCCTRACE(SCMOD) << "Got write request to address 0x" << std::hex << gp->get_address();
204  } else
205  SCCTRACE(SCMOD) << "Got read request to address 0x" << std::hex << gp->get_address();
206  auto ext = gp->get_extension<obi::obi_extension>();
207  if(ID_WIDTH && aid_i.get_interface())
208  ext->set_id(aid_i->read());
209  if(USER_WIDTH) {
210  if(auser_i.get_interface())
211  ext->set_auser(auser_i->read());
212  if(wuser_i.get_interface() && we_i.read())
213  ext->set_duser(wuser_i->read());
214  }
215  auto& state = states[gp.get()];
216  phase_type phase = tlm::BEGIN_REQ;
217  auto delay = sc_core::SC_ZERO_TIME;
218  auto ret = isckt->nb_transport_fw(*gp, phase, delay);
219  auto startResp = false;
220  state.last_phase = phase;
221  if(ret == tlm::TLM_ACCEPTED) {
222  SCCTRACE(SCMOD) << "waiting for TLM reponse";
223  auto resp = achannel_rsp.get();
224  sc_assert(resp.get() == gp.get());
225  }
226  unsigned gnt_delay = req2gnt_delay < 0 ? scc::MT19937::uniform(0, -req2gnt_delay) : req2gnt_delay;
227  for(unsigned i = 0U; i < gnt_delay; ++i)
228  wait(clk_i.posedge_event());
229  gnt_o.write(true);
230  wait(clk_i.posedge_event());
231  state.addrPhaseFinished = true;
232  if(state.last_phase == tlm::BEGIN_RESP) {
233  unsigned resp_delay =
234  addr2data_delay < 0 ? scc::MT19937::uniform(0, -addr2data_delay) : addr2data_delay;
235  if(resp_delay) {
236  rchannel_pending_rsp.push_back({gp, resp_delay - 1});
237  } else
238  rchannel_rsp.notify(gp);
239  } else {
240  state.pending_tx = gp;
241  }
242  }
243  }
244 }

246 template <unsigned int DATA_WIDTH, unsigned int ADDR_WIDTH, unsigned int ID_WIDTH, unsigned int USER_WIDTH>
247 inline void target<DATA_WIDTH, ADDR_WIDTH, ID_WIDTH, USER_WIDTH>::rchannel_rsp_t() {
248  rdata_o.write(0);
249  err_o.write(false);
250  if(ID_WIDTH && r_id_o.get_interface())
251  r_id_o->write(0);
252  if(USER_WIDTH && ruser_o.get_interface())
253  ruser_o->write(0);
254  while(true) {
255  rvalid_o.write(false);
256  tlm::scc::tlm_gp_shared_ptr tx = rchannel_rsp.get();
257  if(tx->get_command() == tlm::TLM_READ_COMMAND) {
258  auto offset = tx->get_address() % (DATA_WIDTH / 8);
259  auto rx_data{sc_dt::sc_uint<DATA_WIDTH>(0)};
260  for(unsigned i = 0U, j = offset * 8; i < tx->get_data_length(); ++i, j += 8) {
261  rx_data.range(j + 7, j) = *(tx->get_data_ptr() + i);
262  }
263  rdata_o.write(rx_data);
264  SCCTRACE(SCMOD) << "responding to read request to address 0x" << std::hex << tx->get_address();
265  } else
266  SCCTRACE(SCMOD) << "responding to write request to address 0x" << std::hex << tx->get_address();
267  err_o.write(tx->get_response_status() != tlm::TLM_OK_RESPONSE);
268  if(ID_WIDTH || USER_WIDTH) {
269  auto ext = tx->get_extension<obi::obi_extension>();
270  if(ID_WIDTH && r_id_o.get_interface())
271  r_id_o->write(ext->get_id());
272  if(USER_WIDTH && ruser_o.get_interface())
273  ruser_o->write(ext->get_duser());
274  }
275  rvalid_o.write(true);
276  wait(sample_delay); // let rready settle
277  while(!rready_i.read())
278  wait(rready_i.value_changed_event());
279  phase_type phase = tlm::END_RESP;
280  auto delay = sc_core::SC_ZERO_TIME;
281  auto ret = isckt->nb_transport_fw(*tx, phase, delay);
282  auto it = states.find(tx.get());
283  sc_assert(it != states.end());
284  states.erase(it);
285  wait(clk_i.posedge_event());
286  }
