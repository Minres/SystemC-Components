46  static ftr_writer<COMPRESSED>* db;
47  static void dbCb(const scv_tr_db& _scv_tr_db, scv_tr_db::callback_reason reason, void* data) {
48  // This is called from the scv_tr_db ctor.
49  static string fName("DEFAULT_scv_tr_cbor");
50  switch(reason) {
51  case scv_tr_db::CREATE:
52  if((_scv_tr_db.get_name() != nullptr) && (strlen(_scv_tr_db.get_name()) != 0))
53  fName = _scv_tr_db.get_name();
54  try {
55  db = new ftr_writer<COMPRESSED>(fName+".ftr");
56  } catch(...) {
57  _scv_message::message(_scv_message::TRANSACTION_RECORDING_INTERNAL, "Can't open recording file");
58  }
59  if(!db->cw.enc.ofs.is_open()) {
60  delete db;
61  db=nullptr;
62  } else {
63  double secs = sc_core::sc_time::from_value(1ULL).to_seconds();
64  auto exp = rint(log(secs)/log(10.0));
65  db->writeInfo(static_cast<int8_t>(exp));
66  }
67  break;
68  case scv_tr_db::DELETE:
69  try {
70  delete db;
71  } catch(...) {
72  _scv_message::message(_scv_message::TRANSACTION_RECORDING_INTERNAL, "Can't close recording file");
73  }
74  break;
75  default:
76  _scv_message::message(_scv_message::TRANSACTION_RECORDING_INTERNAL, "Unknown reason in scv_tr_db callback");
77  }
78  }

115  static inline std::string get_name(const char* prefix, const scv_extensions_if* my_exts_p) {
116  string name{prefix};
117  if(!prefix || strlen(prefix) == 0) {
118  name = my_exts_p->get_name();
119  } else {
120  if((my_exts_p->get_name() != nullptr) && (strlen(my_exts_p->get_name()) > 0)) {
121  name += ".";
122  name += my_exts_p->get_name();
123  }
124  }
125  return (name == "") ? "<unnamed>" : name;
126  }

128  static void recordAttributes(uint64_t id, event_type eventType, char const* prefix, const scv_extensions_if* my_exts_p) {
129  if(!db || my_exts_p == nullptr)
130  return;
131  auto name = get_name(prefix, my_exts_p);
132  switch(my_exts_p->get_type()) {
133  case scv_extensions_if::RECORD: {
134  int num_fields = my_exts_p->get_num_fields();
135  if(num_fields > 0) {
136  for(int field_counter = 0; field_counter < num_fields; field_counter++) {
137  const scv_extensions_if* field_data_p = my_exts_p->get_field(field_counter);
138  recordAttributes(id, eventType, prefix, field_data_p);
139  }
140  }
141  } break;
142  case scv_extensions_if::ENUMERATION:
143  recordAttribute(id, eventType, name, ftr::data_type::ENUMERATION,
144  my_exts_p->get_enum_string((int)(my_exts_p->get_integer())));
145  break;
146  case scv_extensions_if::BOOLEAN:
147  recordAttribute(id, eventType, name, ftr::data_type::BOOLEAN, my_exts_p->get_bool());
148  break;
149  case scv_extensions_if::INTEGER:
150  case scv_extensions_if::FIXED_POINT_INTEGER:
151  recordAttribute(id, eventType, name, ftr::data_type::INTEGER, my_exts_p->get_integer());
152  break;
153  case scv_extensions_if::UNSIGNED:
154  recordAttribute(id, eventType, name, ftr::data_type::UNSIGNED, my_exts_p->get_integer());
155  break;
156  case scv_extensions_if::POINTER:
157  recordAttribute(id, eventType, name, ftr::data_type::POINTER, (long long)my_exts_p->get_pointer());
158  break;
159  case scv_extensions_if::STRING:
160  recordAttribute(id, eventType, name, ftr::data_type::STRING, my_exts_p->get_string());
161  break;
162  case scv_extensions_if::FLOATING_POINT_NUMBER:
163  recordAttribute(id, eventType, name, ftr::data_type::FLOATING_POINT_NUMBER, my_exts_p->get_double());
164  break;
165  case scv_extensions_if::BIT_VECTOR: {
166  sc_bv_base tmp_bv(my_exts_p->get_bitwidth());
167  my_exts_p->get_value(tmp_bv);
168  recordAttribute(id, eventType, name, ftr::data_type::BIT_VECTOR, tmp_bv.to_string());
169  } break;
170  case scv_extensions_if::LOGIC_VECTOR: {
171  sc_lv_base tmp_lv(my_exts_p->get_bitwidth());
172  my_exts_p->get_value(tmp_lv);
173  recordAttribute(id, eventType, name, ftr::data_type::LOGIC_VECTOR, tmp_lv.to_string());
174  } break;
175  case scv_extensions_if::ARRAY:
176  for(int array_elt_index = 0; array_elt_index < my_exts_p->get_array_size(); array_elt_index++) {
177  const scv_extensions_if* field_data_p = my_exts_p->get_array_elt(array_elt_index);
178  recordAttributes(id, eventType, prefix, field_data_p);
179  }
180  break;
181  default: {
182  std::array<char, 100> tmpString;
183  sprintf(tmpString.data(), "Unsupported attribute type = %d", my_exts_p->get_type());
184  _scv_message::message(_scv_message::TRANSACTION_RECORDING_INTERNAL, tmpString.data());
185  }
186  }
187  }

199  static void transactionCb(const scv_tr_handle& t, scv_tr_handle::callback_reason reason, void* data) {
200  if(!db ||
201  !t.get_scv_tr_stream().get_scv_tr_db() ||
202  !t.get_scv_tr_stream().get_scv_tr_db()->get_recording())
203  return;
204  uint64_t id = t.get_id();
205  switch(reason) {
206  case scv_tr_handle::BEGIN: {
207  db->startTransaction(id, t.get_scv_tr_generator_base().get_id(),
208  t.get_scv_tr_generator_base().get_scv_tr_stream().get_id(), t.get_begin_sc_time()/sc_core::sc_time(1, sc_core::SC_PS));
209 
210  auto my_exts_p = t.get_begin_exts_p();
211  if(my_exts_p == nullptr)
212  my_exts_p = t.get_scv_tr_generator_base().get_begin_exts_p();
213  if(my_exts_p) {
214  auto tmp_str = t.get_scv_tr_generator_base().get_begin_attribute_name() ?
215  t.get_scv_tr_generator_base().get_begin_attribute_name() : "";
216  recordAttributes(id, event_type::BEGIN, tmp_str, my_exts_p);
217  }
218  } break;
219  case scv_tr_handle::END: {
220  auto my_exts_p = t.get_end_exts_p();
221  if(my_exts_p == nullptr)
222  my_exts_p = t.get_scv_tr_generator_base().get_end_exts_p();
223  if(my_exts_p) {
224  auto tmp_str = t.get_scv_tr_generator_base().get_end_attribute_name() ?
225  t.get_scv_tr_generator_base().get_end_attribute_name() : "";
226  recordAttributes(id, event_type::END, tmp_str, my_exts_p);
227  }
228  db->endTransaction(id, t.get_end_sc_time()/sc_core::sc_time(1, sc_core::SC_PS));
229  } break;
230  default:;
231  }
232  }
233  // ----------------------------------------------------------------------------
234  static void attributeCb(const scv_tr_handle& t, const char* name, const scv_extensions_if* ext, void* data) {
235  if(!db ||
236  ! t.get_scv_tr_stream().get_scv_tr_db()||
237  ! t.get_scv_tr_stream().get_scv_tr_db()->get_recording())
238  return;
239  recordAttributes(t.get_id(), event_type::RECORD, name == nullptr ? "" : name, ext);
240  }
241  // ----------------------------------------------------------------------------
242  static void relationCb(const scv_tr_handle& tr_1, const scv_tr_handle& tr_2, void* data,
243  scv_tr_relation_handle_t relation_handle) {
244  auto& stream1 = tr_1.get_scv_tr_stream();
245  auto txdb = stream1.get_scv_tr_db();
246  if(!db || !txdb || !txdb->get_recording())
247  return;
248  try {
249  auto& stream2 = tr_2.get_scv_tr_stream();
250  db->writeRelation(txdb->get_relation_name(relation_handle), stream1.get_id(), tr_1.get_id(), stream2.get_id(), tr_2.get_id());
251  } catch(std::runtime_error& e) {
252  _scv_message::message(_scv_message::TRANSACTION_RECORDING_INTERNAL, "Can't create transaction relation");
253  }
254  }
255 };
