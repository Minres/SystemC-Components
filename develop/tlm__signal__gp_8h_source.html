289  for(unsigned int i = 0; i < m_extensions.size(); i++) {

298 inline void tlm_generic_payload_base::copy_extensions_from(const tlm_generic_payload_base& other) {
299  // deep copy extensions (sticky and non-sticky)
300  if(m_extensions.size() < other.m_extensions.size())
301  m_extensions.expand(other.m_extensions.size());
302  for(unsigned int i = 0; i < other.m_extensions.size(); i++) {
303  if(other.m_extensions[i]) { // original has extension i
304  if(!m_extensions[i]) { // We don't: clone.
305  tlm_extension_base* ext = other.m_extensions[i]->clone();
306  if(ext) { // extension may not be clonable.
307  if(has_mm()) { // mm can take care of removing cloned extensions
308  set_auto_extension(i, ext);
309  } else { // no mm, user will call free_all_extensions().
310  set_extension(i, ext);
311  }
312  }
313  } else { // We already have such extension. Copy original over it.
314  m_extensions[i]->copy_from(*other.m_extensions[i]);
315  }
316  }
317  }
