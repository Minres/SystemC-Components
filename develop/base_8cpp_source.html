53 fsm_handle* base::find_or_create(payload_type* gp, bool ace) {
54  const static std::array<std::string, 3> cmd{{"RD", "WR", "IGN"}};
55  auto it = active_fsm.find(gp);
56  if(!gp || it == active_fsm.end()) {
57  if(gp) {
58  SCCTRACE(instance_name) << "creating fsm for trans " << *gp;
59  } else {
60  SCCTRACE(instance_name) << "creating fsm for undefined transaction";
61  }
62  if(idle_fsm.empty()) {
63  auto fsm_hndl = create_fsm_handle();
64  auto fsm = new AxiProtocolFsm();
65  fsm->initiate();
66  fsm_hndl->fsm = fsm;
67  setup_callbacks(fsm_hndl);
68  idle_fsm.push_back(fsm_hndl);
69  }
70  auto fsm_hndl = idle_fsm.front();
71  idle_fsm.pop_front();
72  fsm_hndl->reset();
73  if(gp != nullptr) {
74  fsm_hndl->trans = gp;
75  } else {
76  fsm_hndl->trans = ace ? tlm::scc::tlm_mm<>::get().allocate<ace_extension>()
77  : tlm::scc::tlm_mm<>::get().allocate<axi4_extension>();
78  }
79  active_fsm.insert(std::make_pair(fsm_hndl->trans.get(), fsm_hndl));
80  fsm_hndl->start = sc_time_stamp();
81  return fsm_hndl;
82  } else {
83  it->second->start = sc_time_stamp();
84  return it->second;
85  }
86 }

88 void base::process_fsm_event() {
89  while(auto e = fsm_event_queue.get_next()) {
90  auto entry = e.get();

92  schedule(std::get<0>(entry), std::get<1>(entry), 0);
93  else

98 void base::process_fsm_clk_queue() {
99  if(!fsm_clk_queue_hndl.valid())
100  fsm_clk_queue_hndl = sc_process_handle(sc_get_current_process_handle());
101  if(fsm_clk_queue.avail())
102  while(fsm_clk_queue.avail()) {
103  auto entry = fsm_clk_queue.front();
104  if(std::get<2>(entry) == 0) {
105  SCCTRACE(instance_name) << "processing event " << evt2str(std::get<0>(entry)) << " of trans "
106  << *std::get<1>(entry);
107  react(std::get<0>(entry), std::get<1>(entry));
108  } else {
109  std::get<2>(entry) -= 1;
110  fsm_clk_queue.push_back(entry);
111  }
112  fsm_clk_queue.pop_front();
113  }
114  else
115  // fall asleep if there is nothing to process
116  fsm_clk_queue_hndl.disable();
117 }

119 void base::react(protocol_time_point_e event, axi::fsm::fsm_handle* fsm_hndl) {
120  switch(event) {
121  case WValidE:
122  fsm_hndl->fsm->process_event(WReq());
123  return;
124  case WReadyE:
125  case RequestPhaseBeg:
126  if(is_burst(*fsm_hndl->trans) && fsm_hndl->trans->is_write() &&
127  !is_dataless(fsm_hndl->trans->get_extension<axi::ace_extension>()))
128  fsm_hndl->fsm->process_event(BegPartReq());
129  else
130  fsm_hndl->fsm->process_event(BegReq());
131  return;
132  case BegPartReqE:
133  fsm_hndl->fsm->process_event(BegPartReq());
134  return;
135  case EndPartReqE:
136  fsm_hndl->fsm->process_event(EndPartReq());
137  return;
138  case BegReqE:
139  fsm_hndl->fsm->process_event(BegReq());
140  return;
141  case EndReqE:
142  fsm_hndl->fsm->process_event(EndReq());
143  return;
144  case BegPartRespE:
145  fsm_hndl->fsm->process_event(BegPartResp());
146  return;
147  case EndPartRespE:
148  fsm_hndl->fsm->process_event(EndPartResp());
149  return;
150  case BegRespE:
151  fsm_hndl->fsm->process_event(BegResp());
152  return;
153  case EndRespE:
154  if(!coherent || fsm_hndl->is_snoop) {
155  SCCTRACE(instance_name) << "freeing fsm for trans " << *fsm_hndl->trans;
156  fsm_hndl->fsm->process_event(EndResp());
157  active_fsm.erase(fsm_hndl->trans.get());
158  fsm_hndl->trans = nullptr;
159  idle_fsm.push_back(fsm_hndl);
160  finish_evt.notify();
161  } else {
162  fsm_hndl->fsm->process_event(EndRespNoAck());
163  }
164  return;
165  case Ack:
166  SCCTRACE(instance_name) << "freeing fsm for trans " << *fsm_hndl->trans;
167  fsm_hndl->fsm->process_event(AckRecv());
168  active_fsm.erase(fsm_hndl->trans.get());
169  fsm_hndl->trans = nullptr;
170  idle_fsm.push_back(fsm_hndl);
171  finish_evt.notify();
172  return;
173  default:
174  SCCFATAL(instance_name) << "No valid protocol time point";
175  }
176 }

178 tlm_sync_enum base::nb_fw(payload_type& trans, phase_type const& phase, sc_time& t) {
179  SCCTRACE(instance_name) << "base::nb_fw " << phase << " of trans " << trans;
180  if(phase == BEGIN_PARTIAL_REQ || phase == BEGIN_REQ) { // read/write
181  auto fsm = find_or_create(&trans);
182  if(!trans.is_read()) {
183  protocol_time_point_e evt = axi::fsm::RequestPhaseBeg;
184  if(fsm->beat_count == 0 && wr_start != RequestPhaseBeg)
185  evt = wr_start;
186  else
187  evt = phase == BEGIN_PARTIAL_REQ ? BegPartReqE : BegReqE;
188  if(t == SC_ZERO_TIME) {
189  react(evt, &trans);
190  } else {
191  schedule(evt, &trans, t);
192  }
193  } else {
194  if(t == SC_ZERO_TIME) {
195  react(BegReqE, &trans);
196  } else
197  schedule(BegReqE, &trans, t);
198  }
199  } else if(phase == END_PARTIAL_RESP || phase == END_RESP) {
200  if(t == SC_ZERO_TIME) {
201  react(phase == END_RESP ? EndRespE : EndPartRespE, &trans);
202  } else
203  schedule(phase == END_RESP ? EndRespE : EndPartRespE, &trans, t);
204  } else if(phase == END_REQ) { // snoop access resp
205  if(t == SC_ZERO_TIME) {
206  react(EndReqE, &trans);
207  } else
208  schedule(EndReqE, &trans, t);
209  } else if(phase == BEGIN_PARTIAL_RESP || phase == BEGIN_RESP) { // snoop access resp
210  if(t == SC_ZERO_TIME) {
211  react(phase == BEGIN_RESP ? BegRespE : BegPartRespE, &trans);
212  } else
213  schedule(phase == BEGIN_RESP ? BegRespE : BegPartRespE, &trans, t);
214  } else if(phase == axi::ACK) {
215  if(t == SC_ZERO_TIME) {
216  react(Ack, &trans);
217  } else
218  schedule(Ack, &trans, t);
219  }
220  return TLM_ACCEPTED;
221 }

223 tlm_sync_enum base::nb_bw(payload_type& trans, phase_type const& phase, sc_time& t) {
224  SCCTRACE(instance_name) << "base::nb_bw " << phase << " of trans " << trans;
225  if(phase == END_PARTIAL_REQ || phase == END_REQ) { // read/write
226  if(t == SC_ZERO_TIME) {
227  react(phase == END_REQ ? EndReqE : EndPartReqE, &trans);
228  } else
229  schedule(phase == END_REQ ? EndReqE : EndPartReqE, &trans, t);
230  } else if(phase == BEGIN_PARTIAL_RESP || phase == BEGIN_RESP) { // read/write response
231  if(t == SC_ZERO_TIME) {
232  react(phase == BEGIN_RESP ? BegRespE : BegPartRespE, &trans);
233  } else
234  schedule(phase == BEGIN_RESP ? BegRespE : BegPartRespE, &trans, t);
235  } else if(phase == BEGIN_REQ) { // snoop read
236  auto fsm_hndl = find_or_create(&trans, true);
237  fsm_hndl->is_snoop = true;
238  if(t == SC_ZERO_TIME) {
239  react(BegReqE, &trans);
240  } else
241  schedule(BegReqE, &trans, t);
242  } else if(phase == END_PARTIAL_RESP || phase == END_RESP) { // snoop read response
243  if(t == SC_ZERO_TIME) {
244  react(phase == END_RESP ? EndRespE : EndPartRespE, &trans);
245  } else
246  schedule(phase == END_RESP ? EndRespE : EndPartRespE, &trans, t);
247  }
248  return TLM_ACCEPTED;
249 }
