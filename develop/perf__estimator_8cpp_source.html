66 void perf_estimator::end_of_simulation() {
67  eos.set();
68  sc_time now = sc_time_stamp();
69  unsigned long long elapsed_wall = (eos.wall_clock_stamp - sos.wall_clock_stamp).total_microseconds();
70  auto elapsed_proc = (unsigned long long)((eos.proc_clock_stamp - sos.proc_clock_stamp) * 1000000);
71  auto elapsed_sim = (unsigned long long)(now.to_seconds() * 1000000.);
72  if(elapsed_sim > 0) {
73  double wall_perf = elapsed_wall / elapsed_sim;
74  double proc_perf = elapsed_proc / elapsed_sim;
75  SCCINFO(SCMOD) << "Wall clock (process clock) based simulation real time factor is " << wall_perf << "("
76  << proc_perf << ")";
77  }
78  get_memory();
79 }

88 auto scc::perf_estimator::time_stamp::get_cpu_time() -> double {
89 #if defined(_WIN32)
90  FILETIME create_time;
91  FILETIME exit_time;
92  FILETIME kernel_time;
93  FILETIME user_time;
94  if(GetProcessTimes(GetCurrentProcess(), &create_time, &exit_time, &kernel_time, &user_time) != -1) {
95  SYSTEMTIME system_time;
96  if(FileTimeToSystemTime(&user_time, &system_time) != -1)
97  return (double)system_time.wHour * 3600.0 + (double)system_time.wMinute * 60.0 +
98  (double)system_time.wSecond + (double)system_time.wMilliseconds / 1000.;
99  }
100 #elif defined(__unix__) || defined(__unix) || defined(unix) || (defined(__MACH__) && defined(__APPLE__))
101 #if _POSIX_TIMERS > 0
102  {
103  clockid_t id;
104  struct timespec stamp {};
105 #if _POSIX_CPUTIME > 0
106  if(clock_getcpuclockid(0, &id) == -1)
107 #endif
108 #if defined(CLOCK_PROCESS_CPUTIME_ID)
109  id = CLOCK_PROCESS_CPUTIME_ID;
110 #elif defined(CLOCK_VIRTUAL)
111  id = CLOCK_VIRTUAL;
112 #else
113  id = (clockid_t)-1;
114 #endif
115  if(id != (clockid_t)-1 && clock_gettime(id, &stamp) != -1)
116  return (double)stamp.tv_sec + (double)stamp.tv_nsec / 1000000000.0;
117  }
118 #endif
119 #if defined(RUSAGE_SELF)
120  {
121  struct rusage usage {};
122  if(getrusage(RUSAGE_SELF, &usage) != -1)
123  return (double)usage.ru_utime.tv_sec + (double)usage.ru_utime.tv_usec / 1000000.0;
124  }
125 #endif
126 #if defined(_SC_CLK_TICK)
127  {
128  const double ticks = (double)sysconf(_SC_CLK_TCK);
129  struct tms s;
130  if(times(&s) != (clock_t)-1)
131  return (double)s.tms_utime / ticks;
132  }
133 #endif
134 #if defined(CLOCKS_PER_SEC)
135  {
136  clock_t c = clock();
137  if(c != (clock_t)-1)
138  return (double)c / (double)CLOCKS_PER_SEC;
139  }
140 #endif
141 #endif
142  return 1.0;
143 }

145 long scc::perf_estimator::get_memory() {
146 #if defined(RUSAGE_SELF)
147  {
148  struct rusage usage {};
149  if(getrusage(RUSAGE_SELF, &usage) != -1) {
150  max_memory = std::max(max_memory, usage.ru_maxrss);
151  return usage.ru_maxrss;
152  }
153  }
154 #endif
155  return 0L;
