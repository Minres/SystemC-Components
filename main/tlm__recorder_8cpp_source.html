27 void record(SCVNS scv_tr_handle& handle, tlm::tlm_generic_payload& o) {
28  handle.record_attribute("trans.address", o.get_address());
29  handle.record_attribute("trans.cmd", cmd2char.at(o.get_command()));
30  handle.record_attribute("trans.data_ptr", o.get_data_ptr());
31  handle.record_attribute("trans.data_length", o.get_data_length());
32  handle.record_attribute("trans.response", resp2char.at(1-o.get_response_status()));
33  handle.record_attribute("trans.dmi_allowed", o.is_dmi_allowed());
34  handle.record_attribute("trans.byte_enable", o.get_byte_enable_ptr());
35  handle.record_attribute("trans.byte_enable_length", o.get_byte_enable_length());
36  handle.record_attribute("trans.streaming_width", o.get_streaming_width());
37  handle.record_attribute("trans.gp_option", gp_option2char.at(o.get_gp_option()));
38  if(o.get_data_length() < 8 && o.get_data_ptr()) {
39  uint64_t buf = 0;
40  // FIXME: this is endianess dependent
41  for(size_t i = 0; i < o.get_data_length(); i++)
42  buf += (*o.get_data_ptr()) << i * 8;
43  handle.record_attribute("trans.data_value", buf);
44  }
45 }
46 void record(SCVNS scv_tr_handle& handle, tlm::tlm_phase& o) {
