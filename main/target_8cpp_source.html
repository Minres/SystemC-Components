37 template <unsigned WIDTH> void target<WIDTH>::bfm_thread() {
38  tlm::scc::tlm_mm<>& mm = tlm::scc::tlm_mm<>::get();
39  auto const log_width = scc::ilog2(WIDTH / 8);
40  auto beat_cnt = 0U;
41  wait(SC_ZERO_TIME);
42  while(true) {
43  wait();
44  if(!HRESETn_i.read()) {
45  HREADY_o.write(true);
46  data_payload = nullptr;
47  addr_payload = nullptr;
48  } else {
49  if(HSEL_i.read()) {
50  tlm::tlm_generic_payload* gp{nullptr};
51  if(HTRANS_i.read() > 0x1) { // HTRANS/BUSY or IDLE check
52  gp = mm.allocate();
53  gp->acquire();
54  gp->set_address(HADDR_i.read());
55  if(HWRITE_i.read())
56  gp->set_write();
57  else
58  gp->set_read();
59  // gp->set_command(HWRITE_i.read()?tlm::TLM_WRITE_COMMAND:tlm::TLM_READ_COMMAND);
60  auto* ext = new ahb_extension();
61  gp->set_extension(ext);
62  ext->set_locked(HMASTLOCK_i.read());
63  ext->set_protection(HPROT_i.read());
64  ext->set_burst(static_cast<ahb::burst_e>(HBURST_i.read().to_uint()));
65  size_t size = HSIZE_i.read();
66  if(size > log_width)
67  SCCERR(SCMOD) << "Access size (" << size << ") is larger than bus width (" << log_width << ")!";
68  unsigned length = (1 << size) * (1 << static_cast<unsigned>(ext->get_burst()));
69  gp->set_data_length(length);
70  gp->set_streaming_width(length);
71  gp->set_data_ptr(new uint8_t[length]);
72  if(addr_payload)
73  HREADY_o.write(false);
74  else {
75  HREADY_o.write(true);
76  addr_payload = gp;
77  }
78  }
79  if(data_payload && data_payload->is_write())
80  handle_data_phase(beat_cnt);
81  if(gp && gp->is_read()) {
82  sc_time delay;
83  isckt->b_transport(*gp, delay);
84  }
85  if(!data_payload) {
86  data_payload = addr_payload;
87  addr_payload = nullptr;
88  }
89  if(data_payload && data_payload->is_read())
90  handle_data_phase(beat_cnt);
91  }
92  }
93  }
94 }

96 template <unsigned WIDTH> void target<WIDTH>::handle_data_phase(unsigned& beat_cnt) {
97  auto const width = WIDTH / 8;
98  auto* ext = data_payload->get_extension<ahb_extension>();
99  auto start_offs = data_payload->get_address() & (width - 1);
100  sc_assert((start_offs + data_payload->get_data_length()) <= width);
101  data_t data{0};
102  auto offset = width * beat_cnt;
103  auto len = std::min(data_payload->get_data_length(), width);
104  if(data_payload->is_write()) {
105  data = HWDATA_i.read();
106  for(size_t i = start_offs * 8, j = 0; j < len; i += 8, ++j, ++offset)
107  *(uint8_t*)(data_payload->get_data_ptr() + offset) = data.range(i + 7, i).to_uint();
108  } else {
109  for(size_t i = start_offs * 8, j = 0; j < len; i += 8, ++j, ++offset)
110  data.range(i + 7, i) = *(uint8_t*)(data_payload->get_data_ptr() + offset);
111  HRDATA_o.write(data);
112  }
113  if(++beat_cnt == 1 << static_cast<unsigned>(ext->get_burst())) {
114  if(data_payload->is_write()) {
115  sc_time delay;
116  isckt->b_transport(*data_payload, delay);
117  }
118  beat_cnt = 0;
119  data_payload->release();
120  data_payload = nullptr;
121  }
122 }
