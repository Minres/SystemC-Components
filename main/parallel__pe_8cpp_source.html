32 void parallel_pe::transport(tlm::tlm_generic_payload& payload, bool lt_transport) {
33  if(!waiting_ids.size()) {
34  auto id = threads.size();
35  threads.resize(threads.size() + 1);
36  thread_unit& tu = threads.back();
37  tu.hndl = sc_core::sc_spawn(
38  [this, id]() -> void {
39  auto& tu = threads[id];
40  while(true) {
41  fw_o->transport(*tu.gp, tu.lt_transport);
42  bw_o->transport(*tu.gp);
43  if(tu.gp->has_mm())
44  tu.gp->release();
45  tu.gp = nullptr;
46  waiting_ids.push_back(id);
47  wait(tu.evt);
48  assert(tu.gp);
49  }
50  },
51  sc_core::sc_gen_unique_name("execute"));
52  tu.gp = &payload;
53  } else {
54  auto& tu = threads[waiting_ids.front()];
55  waiting_ids.pop_front();
56  tu.gp = &payload;
57  tu.lt_transport = lt_transport;
58  tu.evt.notify();
59  }
60  if(payload.has_mm())
61  payload.acquire();
62 }
