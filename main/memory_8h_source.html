119 template <unsigned long long SIZE, unsigned BUSWIDTH>
120 int memory<SIZE, BUSWIDTH>::handle_operation(tlm::tlm_generic_payload& trans, sc_core::sc_time& delay) {
121  ::sc_dt::uint64 adr = trans.get_address();
122  uint8_t* ptr = trans.get_data_ptr();
123  unsigned len = trans.get_data_length();
124  uint8_t* byt = trans.get_byte_enable_ptr();
125  unsigned wid = trans.get_streaming_width();
126  // check address range and check for unsupported features,
127  // i.e. byte enables, streaming, and bursts
128  // Can ignore DMI hint and extensions
129  if(adr + len > ::sc_dt::uint64(SIZE)) {
130  SC_REPORT_ERROR("TLM-2", "generic payload transaction exceeeds memory size");
131  trans.set_response_status(tlm::TLM_ADDRESS_ERROR_RESPONSE);
132  return 0;
133  }
134  if(wid < len) {
135  SCCERR(SCMOD) << "Streaming width: " << wid << ", data length: " << len;
136  SC_REPORT_ERROR("TLM-2", "generic payload transaction not supported");
137  trans.set_response_status(tlm::TLM_GENERIC_ERROR_RESPONSE);
138  return 0;
139  }
140  if(byt) {
141  auto res = std::accumulate(byt, byt + trans.get_byte_enable_length(), 0xff,
142  [](uint8_t a, uint8_t b) { return a | b; });
143  if(trans.get_byte_enable_length() != len || res != 0xff) {
144  SC_REPORT_ERROR("TLM-2", "generic payload transaction with scattered byte enable not supported");
145  trans.set_response_status(tlm::TLM_GENERIC_ERROR_RESPONSE);
146  return 0;
147  }
148  }
149  tlm::tlm_command cmd = trans.get_command();
150  SCCTRACE(SCMOD) << (cmd == tlm::TLM_READ_COMMAND ? "read" : "write") << " access to addr 0x" << std::hex << adr;
151  if(cmd == tlm::TLM_READ_COMMAND) {
152 #ifdef HAS_CCI
153  delay += rd_resp_delay;
154 #endif
155  if(mem.is_allocated(adr)) {
156  const auto& p = mem(adr / mem.page_size);
157  auto offs = adr & mem.page_addr_mask;
158  if((offs + len) > mem.page_size) {
159  auto first_part = mem.page_size - offs;
160  std::copy(p.data() + offs, p.data() + offs + first_part, ptr);
161  const auto& p2 = mem((adr / mem.page_size) + 1);
162  std::copy(p2.data(), p2.data() + len - first_part, ptr + first_part);
163  } else {
164  std::copy(p.data() + offs, p.data() + offs + len, ptr);
165  }
166  } else {
167  // no allocated page so return randomized data
168  for(size_t i = 0; i < len; i++)
169  ptr[i] = scc::MT19937::uniform() % 256;
170  }
171  } else if(cmd == tlm::TLM_WRITE_COMMAND) {
172 #ifdef HAS_CCI
173  delay += wr_resp_delay;
174 #endif
175  auto& p = mem(adr / mem.page_size);
176  auto offs = adr & mem.page_addr_mask;
177  if((offs + len) > mem.page_size) {
178  auto first_part = mem.page_size - offs;
179  std::copy(ptr, ptr + first_part, p.data() + offs);
180  auto& p2 = mem((adr / mem.page_size) + 1);
181  std::copy(ptr + first_part, ptr + len, p2.data());
182  } else {
183  std::copy(ptr, ptr + len, p.data() + offs);
184  }
185  }
186  trans.set_response_status(tlm::TLM_OK_RESPONSE);
187  trans.set_dmi_allowed(true);
188  return len;
189 }

191 template <unsigned long long SIZE, unsigned BUSWIDTH>
192 inline bool memory<SIZE, BUSWIDTH>::handle_dmi(tlm::tlm_generic_payload& gp, tlm::tlm_dmi& dmi_data) {
193  auto& p = mem(gp.get_address() / mem.page_size);
194  dmi_data.set_start_address(gp.get_address() & ~mem.page_addr_mask);
195  // TODO: fix to provide the correct end address
196  dmi_data.set_end_address(dmi_data.get_start_address() + mem.page_size - 1);
197  dmi_data.set_dmi_ptr(p.data());
198  dmi_data.set_granted_access(tlm::tlm_dmi::DMI_ACCESS_READ_WRITE);
199 #ifdef HAS_CCI
200  dmi_data.set_read_latency(rd_resp_delay.get_value());
201  dmi_data.set_write_latency(wr_resp_delay.get_value());
202 #endif
203  return true;
204 }
