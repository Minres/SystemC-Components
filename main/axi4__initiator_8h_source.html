105 template <typename CFG> inline void axi::pin::axi4_initiator<CFG>::write_ar(tlm::tlm_generic_payload& trans) {
106  sc_dt::sc_uint<CFG::ADDRWIDTH> addr = trans.get_address();
107  this->ar_addr.write(addr);
108  if(auto ext = trans.get_extension<axi::axi4_extension>()) {
109  this->ar_prot.write(ext->get_prot());
110  if(!CFG::IS_LITE) {
111  this->ar_id->write(sc_dt::sc_uint<CFG::IDWIDTH>(ext->get_id()));
112  this->ar_len->write(sc_dt::sc_uint<8>(ext->get_length()));
113  this->ar_size->write(sc_dt::sc_uint<3>(ext->get_size()));
114  this->ar_burst->write(sc_dt::sc_uint<2>(axi::to_int(ext->get_burst())));
115  this->ar_cache->write(sc_dt::sc_uint<4>(ext->get_cache()));
116  this->ar_qos->write(ext->get_qos());
117  if(this->ar_user.get_interface())
118  this->ar_user->write(ext->get_user(axi::common::id_type::CTRL));
119  }
120  }

144 inline void axi::pin::axi4_initiator<CFG>::write_wdata(tlm::tlm_generic_payload& trans, unsigned beat, bool last) {
145  typename CFG::data_t data{0};
146  sc_dt::sc_uint<CFG::BUSWIDTH / 8> strb{0};
147  auto ext = trans.get_extension<axi::axi4_extension>();
148  auto size = 1u << ext->get_size();
149  auto offset = trans.get_address() & (CFG::BUSWIDTH / 8 - 1);
150  auto beptr = trans.get_byte_enable_length() ? trans.get_byte_enable_ptr() + beat * size : nullptr;
151  if(offset && (size + offset) > (CFG::BUSWIDTH / 8)) { // un-aligned multi-beat access
152  if(beat == 0) {
153  auto bptr = trans.get_data_ptr();
154  for(size_t i = offset; i < size; ++i, ++bptr) {
155  auto bit_offs = i * 8;
156  data(bit_offs + 7, bit_offs) = *bptr;
157  if(beptr) {
158  strb[i] = *beptr == 0xff;
159  ++beptr;
160  } else
161  strb[i] = true;
162  }
163  } else {
164  auto beat_start_idx = beat * size - offset;
165  auto data_len = trans.get_data_length();
166  auto bptr = trans.get_data_ptr() + beat_start_idx;
167  for(size_t i = offset; i < size && (beat_start_idx + i) < data_len; ++i, ++bptr) {
168  auto bit_offs = i * 8;
169  data(bit_offs + 7, bit_offs) = *bptr;
170  if(beptr) {
171  strb[i] = *beptr == 0xff;
172  ++beptr;
173  } else
174  strb[i] = true;
175  }
176  }
177  } else { // aligned or single beat access
178  auto bptr = trans.get_data_ptr() + beat * size;
179  for(size_t i = 0; i < size; ++i, ++bptr) {
180  auto bit_offs = (offset + i) * 8;
181  data(bit_offs + 7, bit_offs) = *bptr;
182  if(beptr) {
183  strb[i] = *beptr == 0xff;
184  ++beptr;
185  } else
186  strb[i] = true;
187  }
188  }
189  this->w_data.write(data);
190  this->w_strb.write(strb);
191  if(!CFG::IS_LITE) {
192  this->w_id->write(ext->get_id());
193  if(this->w_user.get_interface())
194  this->w_user->write(ext->get_user(axi::common::id_type::DATA));
195  }

198 template <typename CFG> inline void axi::pin::axi4_initiator<CFG>::setup_callbacks(fsm_handle* fsm_hndl) {
199  fsm_hndl->fsm->cb[RequestPhaseBeg] = [this, fsm_hndl]() -> void {
200  fsm_hndl->beat_count = 0;
201  outstanding_cnt[fsm_hndl->trans->get_command()]++;
202  if(CFG::IS_LITE) {
203  auto offset = fsm_hndl->trans->get_address() % (CFG::BUSWIDTH / 8);
204  if(offset + fsm_hndl->trans->get_data_length() > CFG::BUSWIDTH / 8) {
205  SCCFATAL(SCMOD) << " transaction " << *fsm_hndl->trans << " is not AXI4Lite compliant";
206  }
207  }
208  };
209  fsm_hndl->fsm->cb[BegPartReqE] = [this, fsm_hndl]() -> void {
210  sc_assert(fsm_hndl->trans->is_write());
211  if(fsm_hndl->beat_count == 0) {
212  write_aw(*fsm_hndl->trans);
213  aw_evt.notify(sc_core::SC_ZERO_TIME);
214  }
215  write_wdata(*fsm_hndl->trans, fsm_hndl->beat_count);
216  active_req[tlm::TLM_WRITE_COMMAND] = fsm_hndl;
217  wdata_vl.write(0x1);
218  };
219  fsm_hndl->fsm->cb[EndPartReqE] = [this, fsm_hndl]() -> void {
220  active_req[tlm::TLM_WRITE_COMMAND] = nullptr;
221  tlm::tlm_phase phase = axi::END_PARTIAL_REQ;
222  sc_core::sc_time t; //(clk_if?clk_if->period()-1_ps:sc_core::SC_ZERO_TIME);
223  auto ret = tsckt->nb_transport_bw(*fsm_hndl->trans, phase, t);
224  fsm_hndl->beat_count++;
225  };
226  fsm_hndl->fsm->cb[BegReqE] = [this, fsm_hndl]() -> void {
227  switch(fsm_hndl->trans->get_command()) {
228  case tlm::TLM_READ_COMMAND:
229  active_req[tlm::TLM_READ_COMMAND] = fsm_hndl;
230  write_ar(*fsm_hndl->trans);
231  ar_evt.notify(sc_core::SC_ZERO_TIME);
232  break;
233  case tlm::TLM_WRITE_COMMAND:
234  active_req[tlm::TLM_WRITE_COMMAND] = fsm_hndl;
235  if(fsm_hndl->beat_count == 0) {
236  write_aw(*fsm_hndl->trans);
237  aw_evt.notify(sc_core::SC_ZERO_TIME);
238  }
239  write_wdata(*fsm_hndl->trans, fsm_hndl->beat_count, true);
240  wdata_vl.write(0x3);
241  }
242  };
243  fsm_hndl->fsm->cb[EndReqE] = [this, fsm_hndl]() -> void {
244  switch(fsm_hndl->trans->get_command()) {
245  case tlm::TLM_READ_COMMAND:
246  rd_resp_by_id[axi::get_axi_id(*fsm_hndl->trans)].push_back(fsm_hndl);
247  active_req[tlm::TLM_READ_COMMAND] = nullptr;
248  break;
249  case tlm::TLM_WRITE_COMMAND:
250  wr_resp_by_id[axi::get_axi_id(*fsm_hndl->trans)].push_back(fsm_hndl);
251  active_req[tlm::TLM_WRITE_COMMAND] = nullptr;
252  fsm_hndl->beat_count++;
253  }
254  tlm::tlm_phase phase = tlm::END_REQ;
255  sc_core::sc_time t(sc_core::SC_ZERO_TIME);
256  auto ret = tsckt->nb_transport_bw(*fsm_hndl->trans, phase, t);
257  fsm_hndl->trans->set_response_status(tlm::TLM_OK_RESPONSE);
258  if(auto ext3 = fsm_hndl->trans->get_extension<axi3_extension>()) {
259  ext3->set_resp(resp_e::OKAY);
260  } else if(auto ext4 = fsm_hndl->trans->get_extension<axi4_extension>()) {
261  ext4->set_resp(resp_e::OKAY);
262  } else if(auto exta = fsm_hndl->trans->get_extension<ace_extension>()) {
263  exta->set_resp(resp_e::OKAY);
264  } else
265  sc_assert(false && "No valid AXITLM extension found!");
266  };
267  fsm_hndl->fsm->cb[BegPartRespE] = [this, fsm_hndl]() -> void {
268  // scheduling the response
269  assert(fsm_hndl->trans->is_read());
270  tlm::tlm_phase phase = axi::BEGIN_PARTIAL_RESP;
271  sc_core::sc_time t(sc_core::SC_ZERO_TIME);
272  auto ret = tsckt->nb_transport_bw(*fsm_hndl->trans, phase, t);
273  };
274  fsm_hndl->fsm->cb[EndPartRespE] = [this, fsm_hndl]() -> void {
275  fsm_hndl->beat_count++;
276  r_end_req_evt.notify();
277  };
278  fsm_hndl->fsm->cb[BegRespE] = [this, fsm_hndl]() -> void {
279  // scheduling the response
280  tlm::tlm_phase phase = tlm::BEGIN_RESP;
281  sc_core::sc_time t(sc_core::SC_ZERO_TIME);
282  auto ret = tsckt->nb_transport_bw(*fsm_hndl->trans, phase, t);
283  };
284  fsm_hndl->fsm->cb[EndRespE] = [this, fsm_hndl]() -> void {
285  r_end_req_evt.notify();
286  if(fsm_hndl->trans->is_read())
287  rd_resp_by_id[axi::get_axi_id(*fsm_hndl->trans)].pop_front();
288  if(fsm_hndl->trans->is_write())
289  wr_resp_by_id[axi::get_axi_id(*fsm_hndl->trans)].pop_front();
290  };
291 }

294  while(true) {
295  this->ar_valid.write(false);
296  wait(ar_evt);
297  this->ar_valid.write(true);
298  do {
299  wait(this->ar_ready.posedge_event() | clk_delayed);
300  if(this->ar_ready.read())
301  react(axi::fsm::protocol_time_point_e::EndReqE, active_req[tlm::TLM_READ_COMMAND]);
302  } while(!this->ar_ready.read());
303  wait(clk_i.posedge_event());

307 template <typename CFG> inline void axi::pin::axi4_initiator<CFG>::r_t() {
308  this->r_ready.write(false);
309  while(true) {
310  wait(this->r_valid.posedge_event() | clk_i.negedge_event());
311  if(this->r_valid.event() || (!active_resp[tlm::TLM_READ_COMMAND] && this->r_valid.read())) {
312  auto id = CFG::IS_LITE ? 0U : this->r_id->read().to_uint();
313  auto data = this->r_data.read();
314  auto resp = this->r_resp.read();
315  auto& q = rd_resp_by_id[id];
316  sc_assert(q.size());
317  auto* fsm_hndl = q.front();
318  auto beat_count = fsm_hndl->beat_count;
319  auto size = axi::get_burst_size(*fsm_hndl->trans);
320  auto offset = fsm_hndl->trans->get_address() & (CFG::BUSWIDTH / 8 - 1);
321  if(offset && (size + offset) > (CFG::BUSWIDTH / 8)) { // un-aligned multi-beat access
322  if(beat_count == 0) {
323  auto bptr = fsm_hndl->trans->get_data_ptr();
324  for(size_t i = offset; i < size; ++i, ++bptr) {
325  auto bit_offs = i * 8;
326  *bptr = data(bit_offs + 7, bit_offs).to_uint();
327  }
328  } else {
329  auto beat_start_idx = beat_count * size - offset;
330  auto data_len = fsm_hndl->trans->get_data_length();
331  auto bptr = fsm_hndl->trans->get_data_ptr() + beat_start_idx;
332  for(size_t i = offset; i < size && (beat_start_idx + i) < data_len; ++i, ++bptr) {
333  auto bit_offs = i * 8;
334  *bptr = data(bit_offs + 7, bit_offs).to_uint();
335  }
336  }
337  } else { // aligned or single beat access
338  auto bptr = fsm_hndl->trans->get_data_ptr() + beat_count * size;
339  for(size_t i = 0; i < size; ++i, ++bptr) {
340  auto bit_offs = (offset + i) * 8;
341  *bptr = data(bit_offs + 7, bit_offs).to_uint();
342  }
343  }
344  axi::axi4_extension* e;
345  fsm_hndl->trans->get_extension(e);
346  e->set_resp(axi::into<axi::resp_e>(resp));
347  e->add_to_response_array(*e);
348  auto tp = CFG::IS_LITE || this->r_last->read() ? axi::fsm::protocol_time_point_e::BegRespE
349  : axi::fsm::protocol_time_point_e::BegPartRespE;
350  react(tp, fsm_hndl);
351  wait(r_end_req_evt);
352  this->r_ready.write(true);
353  wait(clk_i.posedge_event());
354  this->r_ready.write(false);
355  }
356  }

371 template <typename CFG> inline void axi::pin::axi4_initiator<CFG>::wdata_t() {
372  while(true) {
373  this->w_valid.write(false);
374  if(!CFG::IS_LITE)
375  this->w_last->write(false);
376  wait(wdata_vl.default_event());
377  auto val = wdata_vl.read();
378  this->w_valid.write(val & 0x1);
379  if(!CFG::IS_LITE)
380  this->w_last->write(val & 0x2);
381  do {
382  wait(this->w_ready.posedge_event() | clk_delayed);
383  if(this->w_ready.read()) {
384  auto evt = CFG::IS_LITE || (val & 0x2) ? axi::fsm::protocol_time_point_e::EndReqE
385  : axi::fsm::protocol_time_point_e::EndPartReqE;
386  react(evt, active_req[tlm::TLM_WRITE_COMMAND]);
387  }
388  } while(!this->w_ready.read());
389  wait(clk_i.posedge_event());
390  }

393 template <typename CFG> inline void axi::pin::axi4_initiator<CFG>::b_t() {
394  this->b_ready.write(false);
395  while(true) {
396  wait(this->b_valid.posedge_event() | clk_i.negedge_event());
397  if(this->b_valid.event() || (!active_resp[tlm::TLM_WRITE_COMMAND] && this->b_valid.read())) {
398  auto id = !CFG::IS_LITE ? this->b_id->read().to_uint() : 0U;
399  auto resp = this->b_resp.read();
400  auto& q = wr_resp_by_id[id];
401  sc_assert(q.size());
402  auto* fsm_hndl = q.front();
403  axi::axi4_extension* e;
404  fsm_hndl->trans->get_extension(e);
405  e->set_resp(axi::into<axi::resp_e>(resp));
406  react(axi::fsm::protocol_time_point_e::BegRespE, fsm_hndl);
407  wait(r_end_req_evt);
408  this->b_ready.write(true);
409  wait(clk_i.posedge_event());
410  this->b_ready.write(false);
411  }
